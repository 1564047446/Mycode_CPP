int max_int(int num, ...)
{
	va_list arg;
	va_start(arg, num);
	int ans = 0;
	while(num--)
	{
		int t = va_arg(arg, int);
		if(t > ans)
			ans = t;
	}
	va_end(arg);
	return ans;
}
