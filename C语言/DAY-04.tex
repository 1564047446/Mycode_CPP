 // 二分法
 
 int search(x)
 {
 
    int head = 1, tail = x;
    int mid = (head + x) >> 1;
    while(head <= tail)
    {
        if(mid == x)
        return 1;
        if(mid < x)
        head = mid + 1;
        if(mid < x)
        tail = mid - 1;
        mid = (head + tail) >> 1;
    }
    return 0;
 
 }
